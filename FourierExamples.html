Generating a Sample Signal with Varying Frequency

fs = 10000; %sampling frequency
dt = 1/fs; %DeltaT
F = [repelem(20,0.3*fs),repelem(80,0.4*fs),repelem(120,0.3*fs)]; %Frequency Array
StopTime = length(F)/fs; %seconds
t = (0:dt:StopTime);
amp = 1;
for i= 1:length(t)-1
signal(i) = amp*sin(2*pi*F(i)*t(i));
end;
 
plot(signal)
